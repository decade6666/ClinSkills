Sub MergeExcelFilesWithPicker()
    Dim folderPath As String
    Dim fileName As String
    Dim combinedWorkbook As Workbook
    Dim sourceWorkbook As Workbook
    Dim ws As Worksheet
    Dim indexSheet As Worksheet
    Dim rowNum As Integer
    Dim fd As FileDialog
    
    ' --- 1. 弹出文件夹选择对话框 ---
    Set fd = Application.FileDialog(msoFileDialogFolderPicker)
    With fd
        .Title = "请选择包含 .xlsx 文件的文件夹"
        If .Show = -1 Then
            folderPath = .SelectedItems(1) & "\"
        Else
            Exit Sub
        End If
    End With

    Set combinedWorkbook = ThisWorkbook
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    
    ' --- 2. 准备目录页 ---
    On Error Resume Next
    Set indexSheet = combinedWorkbook.Sheets("目录页")
    If indexSheet Is Nothing Then
        Set indexSheet = combinedWorkbook.Sheets.Add(Before:=combinedWorkbook.Sheets(1))
        indexSheet.Name = "目录页"
    End If
    On Error GoTo 0
    
    indexSheet.Cells.Clear
    indexSheet.Range("A1:C1").Value = Array("序号", "原始文件名", "Sheet名称(点击跳转)")
    indexSheet.Range("A1:C1").Font.Bold = True
    
    ' --- 3. 循环处理文件 ---
    fileName = Dir(folderPath & "*.xlsx")
    rowNum = 2
    
    Do While fileName <> ""
        If fileName <> combinedWorkbook.Name Then
            Set sourceWorkbook = Workbooks.Open(folderPath & fileName)
            
            For Each ws In sourceWorkbook.Worksheets
                ws.Copy After:=combinedWorkbook.Sheets(combinedWorkbook.Sheets.Count)
                Dim newWs As Worksheet
                Set newWs = combinedWorkbook.Sheets(combinedWorkbook.Sheets.Count)
                
                ' 目录页记录
                indexSheet.Cells(rowNum, 1).Value = rowNum - 1
                indexSheet.Cells(rowNum, 2).Value = fileName
                indexSheet.Hyperlinks.Add Anchor:=indexSheet.Cells(rowNum, 3), _
                    Address:="", SubAddress:="'" & newWs.Name & "'!A1", _
                    TextToDisplay:=newWs.Name
                
                ' --- 核心改进：不覆盖 A1 原有文字 ---
                Dim originalText As String
                originalText = newWs.Range("A1").Value
                
                ' 如果 A1 是空的，直接写 [返回目录]
                If Trim(originalText) = "" Then
                    newWs.Hyperlinks.Add Anchor:=newWs.Range("A1"), _
                        Address:="", SubAddress:="'" & indexSheet.Name & "'!A1", _
                        TextToDisplay:="[返回目录]"
                Else
                    ' 如果 A1 有内容，在内容后面追加，或者直接把超链接赋给整个单元格但不改变文字
                    ' 做法：直接在原 Range 上添加超链接，Address为空，TextToDisplay不填则保留原样
                    newWs.Hyperlinks.Add Anchor:=newWs.Range("A1"), _
                        Address:="", _
                        SubAddress:="'" & indexSheet.Name & "'!A1"
                    ' 提示：此时 A1 的文字会变蓝带下划线，但文字内容依然是 originalText
                End If
                
                rowNum = rowNum + 1
            Next ws
            sourceWorkbook.Close SaveChanges:=False
        End If
        fileName = Dir()
    Loop
    
    indexSheet.Columns("A:C").AutoFit
    indexSheet.Activate
    Application.ScreenUpdating = True
    Application.DisplayAlerts = True
    MsgBox "合并完成！"
End Sub
